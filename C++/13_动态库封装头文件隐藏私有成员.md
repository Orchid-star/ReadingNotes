VS新建空项目，配置类型改为**动态库（dll)**。创造以下.h和.cpp文件，编译输出。

**h文件 CTest.h**

```c++
#pragma once

#ifdef DLL_EXPORT
#define DLL_API __declspec(dllexport)
#else
#define DLL_API __declspec(dllimport)
#endif // DLL_EXPORT

#include <iostream>
using namespace std;

#define CLASS_PRIVATE(x) private:class x##Private *pd;

class DLL_API CTest
{
public:
	CTest();
	void show_ret(string op, double v1, double v2);
	CLASS_PRIVATE(CTest)
};

```

**cpp文件 CTest.cpp**

```c++
#define DLL_EXPORT

class CTestPrivate {
public:
	double m_v1;
	double m_v2;
	void show_plus();
	void show_minus();
	void show_multi();
	void show_div();
};

#include "CTest.h"

CTest::CTest()
	:pd(new CTestPrivate)
{
}

void CTest::show_ret(string op, double v1, double v2)
{
	pd->m_v1 = v1;
	pd->m_v2 = v2;
	if (op == "+") {
		pd->show_plus();
	}
	else if (op == "-") {
		pd->show_minus();
	}
	else if (op == "x") {
		pd->show_multi();
	}
	else if (op == "/") {
		pd->show_div();
	}
}

void CTestPrivate::show_plus()
{
	cout << m_v1 << " + " << m_v1 << " = " << m_v1 + m_v2 << endl;
}

void CTestPrivate::show_minus()
{
	cout << m_v1 << " - " << m_v1 << " = " << m_v1 - m_v2 << endl;
}

void CTestPrivate::show_multi()
{
	cout << m_v1 << " x " << m_v1 << " = " << m_v1 * m_v2 << endl;
}

void CTestPrivate::show_div()
{
	if (m_v2 != 0) {
		cout << m_v1 << " / " << m_v1 << " = " << m_v1 / m_v2 << endl;
	}
	else {
		cout << "failed div" << endl;
	}
}
```



**调用示例**

另建一个工程，将上面动态库工程产生的lib和dll文件拷贝到exe同级目录下。拷贝上面的CTest.h文件，添加到当前工程中。

```c++
#include <iostream>
#include "ctest.h"
#pragma comment(lib, "..\\Release\\Project1.lib")

int main()
{
    std::cout << "Hello World!\n";

    CTest test;

    test.show_ret("+", 20, 45);

}
```
